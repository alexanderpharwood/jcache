jCache.js

A library for local caching in JavaScript

No dependencies - Simple API - Lightweight - Opensource


//Have a play in your browser's console


//Declare your array of items to set

var items = [

    ["string", "I am a string!", null],
    ["int", 99, null],
    ["bool", true, null]

];


//Set the item to cache

jCache.set(items);


//Retreive the item's value from cache

jCache.getValue('string');


//Retreive the item's type from cache

jCache.getType('string');
